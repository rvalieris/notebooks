set.seed(42)
p <-
tibble(
    x=seq(0, 1, length.out=1e3)
) |>
ggplot(aes(y=0,yend=1,x=x,color=x)) +
geom_segment() +
theme_bw() +
coord_radial(expand=FALSE) +
guides(color=guide_colourbar(barheight=15,direction = 'vertical'))
